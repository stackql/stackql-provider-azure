```sql
id,
name,
properties,
type
FROM azure.bot_service.private_link_resources
WHERE resourceGroupName = '{{ resourceGroupName }}' -- required
AND resourceName = '{{ resourceName }}' -- required
AND subscriptionId = '{{ subscriptionId }}' -- required
;
```
