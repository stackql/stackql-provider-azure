```sql
SELECT
primarySharedKey,
secondarySharedKey
FROM azure.log_analytics.shared_keys_shared_keys
WHERE resourceGroupName = '{{ resourceGroupName }}' -- required
AND workspaceName = '{{ workspaceName }}' -- required
AND subscriptionId = '{{ subscriptionId }}' -- required
;
```
