```sql
'{{ properties }}',

'{{ tags }}',
'{{ location }}',

'{{ cloudResourceName }}'

extendedLocation,
location,
properties,
tags
```

```sql

SET 
data__tags = '{{ tags }}'
WHERE 
subscriptionId = '{{ subscriptionId }}' --required
```
