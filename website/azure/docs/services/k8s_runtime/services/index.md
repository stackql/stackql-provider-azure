    values={[

<TabItem value="get">

Get a ServiceResource

```sql
SELECT
properties
FROM azure.k8s_runtime.services
WHERE resourceUri = '{{ resourceUri }}' -- required
AND serviceName = '{{ serviceName }}' -- required
;
```
