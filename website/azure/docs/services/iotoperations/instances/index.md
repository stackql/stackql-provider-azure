```sql
SELECT
extendedLocation,
identity,
location,
properties,
tags
FROM azure.iotoperations.instances
```

Create a InstanceResource

```sql
INSERT INTO azure.iotoperations.instances (
data__properties,
data__extendedLocation,
data__identity,
data__tags,
data__location,
subscriptionId,
resourceGroupName,
instanceName
)
SELECT 
'{{ properties }}',
'{{ extendedLocation }}' /* required */,
'{{ identity }}',
'{{ tags }}',
'{{ location }}',
'{{ subscriptionId }}',
'{{ resourceGroupName }}',
'{{ instanceName }}'
RETURNING
extendedLocation,
identity,
location,
properties,
tags
;
```
