id,
name,
type
FROM azure.cognitive_services.model_capacities
WHERE subscriptionId = '{{ subscriptionId }}' -- required
AND modelFormat = '{{ modelFormat }}' -- required
AND modelName = '{{ modelName }}' -- required
AND modelVersion = '{{ modelVersion }}' -- required
;
