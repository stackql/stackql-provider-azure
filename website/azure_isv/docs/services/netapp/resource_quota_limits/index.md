```sql

properties
FROM azure_isv.netapp.resource_quota_limits
WHERE subscriptionId = '{{ subscriptionId }}' -- required
AND location = '{{ location }}' -- required
AND quotaLimitName = '{{ quotaLimitName }}' -- required
;
```
