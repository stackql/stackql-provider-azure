```sql

etag,
location,
properties,
tags
FROM azure_isv.netapp.snapshot_policies
WHERE subscriptionId = '{{ subscriptionId }}' -- required
AND resourceGroupName = '{{ resourceGroupName }}' -- required
AND accountName = '{{ accountName }}' -- required

;

tags
