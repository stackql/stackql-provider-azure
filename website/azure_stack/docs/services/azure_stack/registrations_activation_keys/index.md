activationKey
FROM azure_stack.azure_stack.registrations_activation_keys
WHERE subscriptionId = '{{ subscriptionId }}' -- required
AND resourceGroup = '{{ resourceGroup }}' -- required
AND registrationName = '{{ registrationName }}' -- required
;
